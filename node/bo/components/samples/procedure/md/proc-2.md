IUD 프로시저 IN/OUT 예시
```
    /***********************************************************************************************/
    /* DEFAULT IN/OUT 변수 선언                                                                    */
    /***********************************************************************************************/
    ot_status_code                  OUT VARCHAR2                      /* 로그상태   Default NULL   */
   ,io_ui_id                     IN OUT VARCHAR2                      /* 화면ID     Default NULL   */
   ,io_prcd_id                   IN OUT VARCHAR2                      /* 프로세스ID Default NULL   */

    /***********************************************************************************************/
    /* 업무적 IN 변수 선언                                                                         */
    /***********************************************************************************************/
    ,in_req_gbn                  IN OUT VARCHAR2                      /* 요청구분(C/U)             */
	,in_chnl_cd                  IN OUT VARCHAR2                      /* 채널_코드                 */
	,in_term_cd                  IN OUT VARCHAR2                      /* 약관_코드                 */
	,in_term_vrsn_no             IN OUT VARCHAR2                      /* 약관_버전_코드              */
	,in_term_patr_dvcd           IN     VARCHAR2                      /* 약관_유형_구분코드           */
	,in_term_nm                  IN     VARCHAR2                      /* 약관_명                  */
	,in_apl_st_dt                IN     VARCHAR2                      /* 적용_시작_일자              */
	,in_apl_ed_dt                IN     VARCHAR2                      /* 적용_종료_일자              */
	,in_term_cntt                IN     VARCHAR2                      /* 약관_내용                 */
	,in_esnt_agrm_term_yn        IN     VARCHAR2                      /* 필수_동의_약관_여부          */

    /***********************************************************************************************/
    /* 사용자 IN 변수 선언                                                                         */
    /***********************************************************************************************/
	,in_user_id                      IN VARCHAR2                      /* 사용자id                  */
    ,in_ip                           IN VARCHAR2                      /* ip                        */

    /***********************************************************************************************/
    /* DEFAULT OUT 변수 선언                                                                       */
    /***********************************************************************************************/
    ,ot_respon_code                  OUT VARCHAR2                     /* 응답코드                  */
    ,ot_res_msg                      OUT VARCHAR2                     /* 응답메시지                */
```
